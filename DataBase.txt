CREATE TABLE books (
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(30),
    price INT,
    stock INT,
    description VARCHAR(300),
    author VARCHAR(50),
    publicationyear VARCHAR(20),
    pages VARCHAR(20),
    src VARCHAR(50),
    PRIMARY KEY (id)
);
CREATE TABLE orders (
    id INT AUTO_INCREMENT PRIMARY KEY,
    price DECIMAL(10,2) NOT NULL,
    bookid INT NOT NULL,
    date DATETIME NOT NULL,
    email VARCHAR(100) NOT NULL
);
CREATE TABLE users (
    id INT NOT NULL AUTO_INCREMENT,
    name VARCHAR(100),
    password VARCHAR(100),
    dob DATE,
    email VARCHAR(100) UNIQUE,
    fiction TINYINT(1),
    self_help TINYINT(1),
    education TINYINT(1),
    PRIMARY KEY (id)
);
